1. Принцип единственной ответственности


class Alcoholic {
	pourDrink() {
		//Налить выпить.
	}
	prepareSnack() {
		//Приготовить закуску.
	}
	drinkUp() {
		//Выпить.
	}
}

Не правильно!

Правильно:


class Barmen {
    constructor() {
        this.pourDrink()
    }

    pourDrink = function () {
        //Налить выпить.
    }
}
								

class Chef {
  constructor() {
      this.prepareSnack()
  }

  prepareSnack = function () {
      //приготовить закуску.
  }
}
								

class Alcoholic {
  constructor() {
    this.niceEvening = [];
    this.drink = new Barmen().pourDrink();
    this.snack = new Chef().prepareSnack();
  }

  drinkUp() {
    this.niceEvening.push(this.drink);
    setTimeout(() => {
      this.niceEvening.push(this.snack);
    }, 2000)
  }
}

Итого: Принцип Единственной ответственности заключается в том, что каждая фунция или модуль отвечает только за-то для чего его сделали
